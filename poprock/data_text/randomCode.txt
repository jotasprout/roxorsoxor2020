SELECT t.*, a.albumNameMB
FROM tracksMB t
JOIN albumsMB a ON a.albumMBID = t.albumMBID
WHERE a.artistMBID = '5182c1d9-c7d2-4dad-afa0-ccfeada921a8'
ORDER BY a.albumNameMB, t.trackNumber ASC;
